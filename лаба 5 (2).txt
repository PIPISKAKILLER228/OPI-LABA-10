#include <iostream>
#include <vector>
using namespace std;

int countPositiveNumbers(const vector<int>& arr) {
    int count = 0;
    for (size_t i = 0; i < arr.size(); ++i) {
        if (arr[i] > 0) {
            ++count;
        }
    }
    return count;
}

int main() {
    vector<int> arr = {-1, 2, 3, -4, 5, 6, -7, 8, 9}; // Пример массива
    int positiveCount = countPositiveNumbers(arr);
    
    cout << "Количество положительных чисел: " << positiveCount << endl;
    return 0;
}
